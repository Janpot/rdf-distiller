def self.open_file(filename_or_url, options = {}, &block)
  STDERR.puts "open file #{filename_or_url.inspect}"
  filename_or_url = $1 if filename_or_url.to_s.match(/^file:(.*)$/)
  Kernel.open(filename_or_url.to_s, &block)
end
