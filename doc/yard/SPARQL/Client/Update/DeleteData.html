def to_s
  query_text = 'DELETE DATA {'
  query_text += ' GRAPH ' + SPARQL::Client.serialize_uri(self.options[:graph]) + ' {' if self.options[:graph]
  query_text += "\n"
  query_text += RDF::NTriples::Writer.buffer { |writer| writer << @data }
  query_text += '}' if self.options[:graph]
  query_text += "}\n"
end
