def to_json
  json = options.dup.to_hash rescue {}
  json.merge!({
    :graph  => graph ? graph.to_uri.to_s : nil,
    :delete => deletes.to_rdf_json,
    :insert => inserts.to_rdf_json,
  })
  json.to_json
end
