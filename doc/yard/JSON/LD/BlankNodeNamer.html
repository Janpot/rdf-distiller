def get_name(old)
  if old && self.has_key?(old)
    self[old]
  elsif old
    self[old] = @prefix.dup
    @prefix.succ!
    self[old]
  else
    # Not referenced, just return a new unique value
    cur = @prefix.dup
    @prefix.succ!
    cur
  end
end
