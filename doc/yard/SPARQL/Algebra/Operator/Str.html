def apply(term)
  case term
    when RDF::Literal then RDF::Literal(term.value)
    when RDF::URI     then RDF::Literal(term.to_s)
    else raise TypeError, "expected an RDF::Literal or RDF::URI, but got #{term.inspect}"
  end
end
